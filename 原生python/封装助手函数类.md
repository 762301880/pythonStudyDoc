import pprint
import sys
def dd(*args):
    """
    类似 Laravel 的 dd() 函数 - 打印变量并终止程序

    参数:
        *args: 一个或多个要打印的变量

    外部使用例
    from helper import dd

    dd(打印即可)
    """
    # 如果没有传递参数，只打印分隔符并退出
    if not args:
        print("=" * 50)
        sys.exit(0)

    # 打印分隔符
    print("=" * 50)

    # 处理每个参数
    for i, arg in enumerate(args):
        # 打印参数索引（如果有多个参数）
        if len(args) > 1:
            print(f"参数 {i + 1}:")

        # 打印变量类型
        print(f"类型: {type(arg).__name__}")

        # 打印变量值（使用pprint进行美观打印）
        print("值:")
        pprint.pprint(arg, indent=4)

        # 打印分隔符
        print("-" * 50)

    # 终止程序
    sys.exit(0)
